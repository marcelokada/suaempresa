==>trede_usuadmin = tabela de usuarios
->rede_sequsua = auto incremente -sequencia do banco
->rede_nomeusu = nome do usuario
->rede_cpfusua = é utilizado par afazer o login
->rede_adminus = se ele é um admin, com a tag "s" ou "n"
->rede_senhaus = senha em md5
->rede_tipousu = nivel de herarquia do 1 ao 5
->rede_usubloc = se o usuario atingir a qtde de logins errados,é bloqueado 
->rede_emailus = email para envio de redefinição de senha             
->rede_serecus = sessao para redefinição de senha         
->rede_drecusu = data e hora da redefinição        
->rede_hrecusu = hora em segundos para verificação, contagem regressiva


==>trede_segtenta = tabela de tentativas de acesso ao banco
->vnumeroipseg = ip da maquina que esta logando
->vnumercpfseg = cpf do usuario que esta logando
->ddatatentseg = data da tentativa
->nnumetentseg = quantidade de tentativas
->vhoratentseg = hora da tentativa
